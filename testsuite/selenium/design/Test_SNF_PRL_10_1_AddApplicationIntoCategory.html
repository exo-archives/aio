Test_SNF_PRL_10_1_AddApplicationIntoCategory
open /portal/public/classic/
echo -AddApplicationToCategory-
click link=Login
waitForElementPresent username
type username root
type password exo
click //div[@id='UIPortalLoginFormAction']/div/div/div
waitForElementPresent link=Application Registry
click link=Application Registry
echo --Add Category
waitForElementPresent //div[@id='UIApplicationOrganizer']//div/div
click //div[@id='UIApplicationOrganizer']//div/div
pause 1000
waitForElementPresent name
type name Test_cat_10
type displayName Test_cat_10
echo --Select permissions
click //div[@class='WorkingArea']//div[@class='TabsContainer']/div[2]//div[@class='MiddleTab']
waitForElementPresent link=Add Permission
click link=Add Permission
waitForElementPresent link=Platform
click link=Platform
waitForElementPresent link=Administrators
click link=Administrators
waitForElementPresent link=manager
click link=manager
waitForTextPresent /platform
click link=Save
waitForElementPresent //div[@id='UIApplicationOrganizer']//a[@class='ControlIcon EditIcon']
waitForTextPresent Test_cat_10
verifyTextPresent Test_cat_10
click link=Test_cat_10
echo --Add application to Test_cat_10
waitForElementPresent //div[@class='ListContent']//div[@class='SelectedTab']//a[@class='ControlIcon CreateNewIcon']
click //div[@class='ListContent']//div[@class='SelectedTab']//a[@class='ControlIcon CreateNewIcon']
echo --Select first application in list
waitForElementPresent //input[@id='displayName']
click //input[@name='application' and @value='1']
type displayName test_displayname_10
click css=form#UIAddApplicationForm div.UIAction div.ActionButton
echo --Delete application
waitForElementPresent //div[@class='ListContent']//div[@class='UIVTabContent']/div[1]//a[@class='ControlIcon DeletePortalIcon']
click //div[@class='ListContent']//div[@class='UIVTabContent']/div[1]//a[@class='ControlIcon DeletePortalIcon']
waitForConfirmation Are you sure to delete this application?
waitForTextNotPresent test_displayname_10
verifyTextNotPresent test_displayname_10
echo --Delete Category
click //div[@class='SelectedTab']//a[@class='ControlIcon DeleteIcon']
assertConfirmation Are you sure to delete this category and all applications on it?
waitForTextNotPresent Test_cat_10
verifyTextNotPresent Test_cat_10
click link=Logout
waitForElementPresent link=Login
