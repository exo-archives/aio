SNF_DMS_06_Copy_Cut_PasteNode
open /portal/public/classic
click link=Login
type username root
type password exo
click link=Sign in
waitForElementPresent link=Sites Explorer
click link=Sites Explorer
waitForElementPresent link=Sites Management
click link=Sites Management
click //div[@id='iconTreeExplorer']//div[2]/a
click link=documents
click link=Add Folder
type name Source_folder_06
click //form[@id='UIFolderForm']/div[2]/div/div/table/tbody/tr/td/div[1]/div/div/div/a
waitForElementPresent link=Source_folder_06
echo --create document
click link=Source_folder_06
click link=Add Document
pause 5000
type name Article_06
type title Article_06_title
click //form[@id='UIDocumentForm']/div[2]/div/table/tbody/tr/td/div[1]/div/div/div/a
waitForElementPresent link=Article_06
echo --copy/paste node
click link=documents
click link=Add Folder
type name Destination_copy_folder_06
click //form[@id='UIFolderForm']/div[2]/div/div/table/tbody/tr/td/div[1]/div/div/div/a
waitForElementPresent link=Destination_copy_folder_06
mouseOver link=Article_06
componentExoContextMenu link=Article_06
click link=Copy
mouseOver link=Destination_folder_06
componentExoContextMenu link=Destination_folder_06
click link=Paste
echo --check after copying
verifyElementPresent link=Article_06
click link=Destination_folder_06
verifyElementPresent link=Article_06
echo --cut/paste node
click link=documents
click link=Add Folder
type name Destination_cut_folder_06
click //form[@id='UIFolderForm']/div[2]/div/div/table/tbody/tr/td/div[1]/div/div/div/a
waitForElementPresent link=Destination_cut_folder_06
click link=Source_folder_06
mouseOver link=Article_06
componentExoContextMenu link=Article_06
click link=Cut
mouseOver link=Destination_cut_folder_06
componentExoContextMenu link=Destination_cut_folder_06
click link=Paste
verifyElementNotPresent link=Article_06
click link=Destination_cut_folder_06
verifyElementPresent link=Article_06
echo --Delete folder, document
mouseOver link=Source_folder_06
componentExoContextMenu link=Source_folder_06
click link=Delete
click //div[@id='UIPopupWindow']/div/div[2]/div/div/div/div/div/div/div/div[2]/table/tbody/tr/td/div[1]/div/div/div/a
verifyElementNotPresent link=Source_folder_06
mouseOver link=Destination__copy_folder_06
componentExoContextMenu link=Destination__copy_folder_06
click link=Delete
click //div[@id='UIPopupWindow']/div/div[2]/div/div/div/div/div/div/div/div[2]/table/tbody/tr/td/div[1]/div/div/div/a
verifyElementNotPresent link=Destination__copy_folder_06
mouseOver link=Destination_cut_folder_06
componentExoContextMenu link=Destination_cut_folder_06
click link=Delete
click //div[@id='UIPopupWindow']/div/div[2]/div/div/div/div/div/div/div/div[2]/table/tbody/tr/td/div[1]/div/div/div/a
verifyElementNotPresent link=Destination_cut_folder_06
click link=Logout
